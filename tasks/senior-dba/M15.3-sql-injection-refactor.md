## Рефакторинг #1 — `update_fields` (sqlite_store.py:419)

### Текущий код
```python
def update_fields(self, memory_id: str, **kwargs: Any) -> bool:
    if not kwargs:
        return False
    allowed = {
        "status", "quality_score", "confidence", "source_coverage",
        "cluster_id", "derived_from", "embedding_id", "clean_content",
        "filter_profile", "filter_stats", "filter_version",
        "title", "content", "tags", "category", "file_path",
    }
    updates = {k: v for k, v in kwargs.items() if k in allowed}
    if not updates:
        return False
    for field in ("derived_from", "tags", "filter_stats"):
        if field in updates and not isinstance(updates[field], str):
            updates[field] = json.dumps(updates[field], ensure_ascii=False)
    updates["updated_at"] = datetime.now(UTC).isoformat()
    setters = ", ".join(f"{k}=?" for k in updates)
    values = [*updates.values(), memory_id]
    conn = self._get_conn()
    cur = conn.execute(f"UPDATE memories SET {setters} WHERE id=?", values)
    return cur.rowcount > 0
```

### Целевой код

```python
# В начале файла, рядом с _MIGRATIONS:

_FIELD_UPDATERS: dict[str, str] = {
    "status": "status=?",
    "quality_score": "quality_score=?",
    "confidence": "confidence=?",
    "source_coverage": "source_coverage=?",
    "cluster_id": "cluster_id=?",
    "derived_from": "derived_from=?",
    "embedding_id": "embedding_id=?",
    "clean_content": "clean_content=?",
    "filter_profile": "filter_profile=?",
    "filter_stats": "filter_stats=?",
    "filter_version": "filter_version=?",
    "title": "title=?",
    "content": "content=?",
    "tags": "tags=?",
    "category": "category=?",
    "file_path": "file_path=?",
}

_JSON_FIELDS: frozenset[str] = frozenset({"derived_from", "tags", "filter_stats"})


class SQLiteStore:
    def update_fields(self, memory_id: str, **kwargs: Any) -> bool:
        """Update arbitrary fields on a memory row.

        Field names must be in _FIELD_UPDATERS whitelist. Values are
        parameterised; JSON fields auto-serialised.
        """
        updates = {k: v for k, v in kwargs.items() if k in _FIELD_UPDATERS}
        if not updates:
            return False
        # Auto-serialise JSON fields
        for field in _JSON_FIELDS & updates.keys():
            if not isinstance(updates[field], str):
                updates[field] = json.dumps(updates[field], ensure_ascii=False)
        setters = ", ".join(_FIELD_UPDATERS[k] for k in updates)
        values = [*updates.values(), datetime.now(UTC).isoformat(), memory_id]
        conn = self._get_conn()
        cur = conn.execute(
            f"UPDATE memories SET {setters}, updated_at=? WHERE id=?",  # nosec B608
            values,
        )
        conn.commit()
        self._invalidate_caches()
        return cur.rowcount > 0
```

`# nosec B608` обоснование в комментарии перед `_FIELD_UPDATERS`:
```python
# _FIELD_UPDATERS is a static whitelist. setters is built by joining
# fragments from this dict — no user input flows into SQL identifiers.
# Values are always parameterised via ? placeholders.
```

## Рефакторинг #2 — `fts_search` (sqlite_store.py:523)

### Текущий код

```python
def fts_search(self, ...):
    sql = f"""
        SELECT ... FROM memories_fts f
        JOIN memories m ON m.id = f.id
        WHERE 1=1 {category_filter} {tag_filter} {project_filter} ...
    """
```

### Целевой код

```python
@staticmethod
def _escape_fts_query(user_query: str) -> str:
    """Sanitise user input for FTS5 MATCH clause.

    Strategy:
    1. Strip FTS5 operators: " ' ( ) * : -
    2. Replace with whitespace
    3. Wrap in double quotes to disable FTS5 syntax
    4. Truncate to max 200 chars (prevent DoS)
    """
    if not user_query:
        return '""'
    # Strip dangerous chars (keep alphanumeric, cyrillic, hyphen within word)
    safe = re.sub(r'[^\w\s\u0400-\u04FF]', " ", user_query)
    # Collapse whitespace
    safe = " ".join(safe.split())
    # Truncate
    if len(safe) > 200:
        safe = safe[:200]
    return f'"{safe}"'  # quoted string disables FTS5 operators


def fts_search(
    self,
    query: str,
    *,
    limit: int = 10,
    project: str | None = None,
    agent: str | None = None,
    status: MemoryStatus | None = None,
    category: str | None = None,
) -> list[Memory]:
    conn = self._get_conn()
    fts_query = self._escape_fts_query(query)
    where_parts = ["memories_fts MATCH ?"]
    params: list[Any] = [fts_query]
    if project:
        where_parts.append("m.project = ?")
        params.append(project)
    if agent:
        where_parts.append("m.agent = ?")
        params.append(agent)
    if status:
        where_parts.append("m.status = ?")
        params.append(status.value)
    if category is not None:
        if category == "__uncategorized":
            where_parts.append("m.category IS NULL")
        else:
            where_parts.append("(m.category = ? OR m.category LIKE ?)")
            params.extend([category, f"{category}/%"])
    where_clause = " AND ".join(where_parts)
    sql = (
        "SELECT m.* FROM memories m "
        "JOIN memories_fts f ON m.rowid = f.rowid "
        f"WHERE {where_clause} "  # nosec B608 — where_parts are static
        "ORDER BY rank LIMIT ?"
    )
    params.append(limit)
    return [self._row_to_memory(r) for r in conn.execute(sql, params).fetchall()]
```

`# nosec B608` обоснование: `where_parts` — статические строки + параметризованные значения, `?` placeholders.
